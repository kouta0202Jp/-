const backgroundImageUrl = 'https://image.itmedia.co.jp/nl/articles/2401/03/matfigmky002.jpg';

function changeBackground() {
    const pageWrapper = document.querySelector('div[data-functional-selector="correct-answer"]');
    const questionWrapper = document.querySelector('main.question__PageWrapper-sc-12j7dwx-2');
    const instructionsWrapper = document.querySelector('main.instructions-with-avatar__PageMainContent-sc-59fa51-1');
    const startWrapper = document.querySelector('main.start__PageMainContent-sc-16554ru-0');
    const sentWrapper = document.querySelector('main.sent__PageWrapper-sc-1s8zmdp-0');
    const incorrectWrapper = document.querySelector('div[data-functional-selector="incorrect-answer"]');
    const preQuestionWrapper = document.querySelector('main.pre-question__PageWrapper-sc-cqz5m6-2');
    const timeUpWrapper = document.querySelector('div[data-functional-selector="time-up-answer"]');
    const resultsPageWrapper = document.querySelector('main[data-functional-selector="results-page"]');

    if (pageWrapper) {
        pageWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        pageWrapper.style.backgroundSize = 'cover';
    }

    if (questionWrapper) {
        questionWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        questionWrapper.style.backgroundSize = 'cover';
    }

    if (instructionsWrapper) {
        instructionsWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        instructionsWrapper.style.backgroundSize = 'cover';
    }

    if (startWrapper) {
        startWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        startWrapper.style.backgroundSize = 'cover';
    }

    if (sentWrapper) {
        sentWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        sentWrapper.style.backgroundSize = 'cover';
    }

    if (incorrectWrapper) {
        incorrectWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        incorrectWrapper.style.backgroundSize = 'cover';
    }

    if (preQuestionWrapper) {
        preQuestionWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        preQuestionWrapper.style.backgroundSize = 'cover';
    }

    if (timeUpWrapper) {
        timeUpWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        timeUpWrapper.style.backgroundSize = 'cover';
    }

    if (resultsPageWrapper) {
        resultsPageWrapper.style.backgroundImage = `url('${backgroundImageUrl}')`;
        resultsPageWrapper.style.backgroundSize = 'cover';
    }
}

// 0.8秒間隔で繰り返し実行
setInterval(changeBackground, 800);
